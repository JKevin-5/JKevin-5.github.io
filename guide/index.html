# 1.1 官网地址

# 1.2 当前博客使用的版本

依赖版本号
node.jsv16.13.0
VuePressv2.0.0-beta.61(last)

在package.json中使用固定版本"vuepress": "2.0.0-beta.61"

# 二、Vuepress config结构

# 1、整体配置内容

export default defineUserConfig({

})

# 2、配置某个页面的侧边栏

使用默认主题。

theme: defaultTheme({
    sidebar: {
        "/画面地址/":[{
            text: "画面名称",
            collapsible: false, // 是否折叠
            children: []
        }]
})

# 3、配置顶部导航栏

使用默认主题。

theme: defaultTheme({
    navbar: [
        '页面路径', // 显示的名称是页面的介绍
        {
            text:'菜单名称',
            children: ['子页面路径']
        }
    ]
})

# 4、配置浏览器标签页icon

export default defineUserConfig({
    ...
    head: [['link', { rel: 'icon', href: 'https://vuejs.org/images/logo.png' }]],
    ...
})
